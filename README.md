Média Aritmética

print('-'*30)
print('AVALIAÇÃO FINAL ')
print('-'*30)

aluno=input('Digite o nome do primeiro aluno:')
n=float(input('Digite a primeira Nota: '))
n1=float(input('Digite a segunda Nota: '))
media=(n+n1) /2
s='Aprovado'
r='Reprovado'
print(aluno,'Teve a média {}'.format(n,n1,media))
if media >= 7:
    print('Situação: ',s)
else:
    print('Situação: ',r)
